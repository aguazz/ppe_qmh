\begin{frame}{What is (regularity) causality}
\vspace{-0.2cm}

\begin{center}
    Hume's \textbf{Regularity Theory of Causality}\footnote{\scriptsize David Hume (1739) A Treatise of Human Nature}$^,$\footnote{    
\scriptsize John Stuart Mill (1843) A System of Logic, Ratiocinative and Inductive: Being a Connected View of the Principles of Evidence, and the Methods of Scientific Investigation}
\end{center}

    \textbf{Principle:} Focuses on Hume's regularity condition. \\
    \textbf{Method:} Empirical observation. \\
    \textbf{Regular Causality:} An event $C$ (cause) always precedes an event $E$ (effect). \\
\vspace{0.6cm}

\begin{center}
    \textbf{Example:} Sunrise and daylight
\end{center}

\textbf{Regularity:} Every time the sun rises ($C$) daylight ($E$) follows. \\
\textbf{Causality:} Therefore sunrise causes daylight.
        
\end{frame}

\begin{frame}{What is (counterfactual) causality?}
\vspace{-0.4cm}

\begin{center}
    David Lewis's \textbf{Counterfactual Causality}\footnote{\scriptsize David K. Lewis (1973) Counterfactuals. The Journal of Symbolic Logic. % Harvard University Press
    }
\end{center}

\textbf{Principle:} Focuses on Hume's necessity condition. \\
\textbf{Actual World:} %The world as it truly exists, 
Events $C$ (presumed cause) and \(E\) (presumed effect) have occurred. \\
\textbf{Possible No-$\pmb{C}$ World:} Hypothetical %, logically coherent 
version of the actual world, where $C$ has not occur %differing only in specific aspects. 
\\
\textbf{Closest No-$\pmb{C}$ Possible World}: The change required to eliminate $C$ is minimal. %, that is, the world remains as similar as possible to the actual world in all other respects. 
\\
\textbf{Counterfactual Causality:} If $E$ does not occur in the closest possible world.
\vspace{0.4cm}

\begin{center}
    \textbf{Example:} A rock breaks a window
\end{center}

\textbf{Actual World:} %The world as it truly exists, 
The rock is thrown ($C$) and the window shatters ($E$)\\
\textbf{Possible No-$\pmb{C}$ World:} The person holding the rock decides not to throw it. \\
\textbf{Closest No-$\pmb{C}$ Possible World}: No other object impacts the window. \\
\textbf{Counterfactual Causality:} The window does not break.

\end{frame}

\begin{frame}{What is (probabilistic) causality?}
\vspace{-0.4cm}

\begin{center}
    Hans Reichenbach's \textbf{Probabilistic Causality}\footnote{\scriptsize Hans Reichenbach (1956) The Direction of Time. The Philosophical Quarterly.
    } \\    
\end{center}

\textbf{Premises:} 
\begin{itemize}
    \item[$\bullet$] Causes not always produce effects \\
    (e.g., smoking does not always cause lung cancer).
    \item[$\bullet$] Effects may have multiple causes \\
    (e.g., lung cancer can result from smoking, pollution, or genetic predisposition).
\end{itemize}

\textbf{Probabilistic Causation:} Event $C$ causes event $E$ if it increases its probability \\
 {
\setlength{\leftmargini}{4.69cm}
\begin{itemize}
    \item[$\bullet$] i.e., $\Pr[E\ |\ C] \geq \Pr[E]$
    \item[$\bullet$] e.g., Smoking raises lung cancer risk
\end{itemize}
}

\textbf{Screening off effect:} A common cause explains the correlation between two events
e.g., Smoking causes both yellow fingers and lung cancer, but yellow fingers does not cause lung cancer or vice versa and.

\end{frame}

\section{\Large The stable unit treatment value assumption}

\begin{frame}{The stable unit treatment value assumption (SUTVA)}

The \textbf{Stable Unit Treatment Value Assumption (SUTVA)} is a key assumption in causal inference, particularly in the Potential Outcome Framework.

    \begin{coolblock}{SUTVA}
        The SUTVA states:
        \begin{itemize}
            \item \textbf{No Interference/Spillovers Assumption (NI):} the potential outcomes for any unit do not vary with the treatments assigned to other units.
            
            \item \textbf{No Hidden Variants (NHV):} for each unit, there are no different forms or versions of each treatment level, which lead to different potential outcomes.
        \end{itemize}
    \end{coolblock}
    \vspace{0.4cm}

    % \textbf{SUTVA means} that the random variables $\lrc{Y_1(T_1)}_{i=1}^n$ are independent and identically distributed to $Y(T)$:
    % \begin{align*}
    %     Y_i(T_i) &\perp Y_j(T_j),\quad \text{ for } i \neq j. \\
    %     Y_i(T_i) &\sim Y(T),\quad\hspace{0.15cm} \text{ for } i = 1,\dots,n.
    % \end{align*}
\end{frame}

\begin{frame}{The SUTVA: no interference/spillovers}

\begin{itemize}
    % \item Let $Y_i(T_i)$ be de outcome of individual $i$ given its treatment $T_i$
    % \item Let $Y_i(T_1,\dots, T_n)$ be the outcome of individual $i$ given its treatment $T_i$ and the treatment administered to all other units, $T_j$ with $j\neq i$.
    \item \textbf{Mathematical interpretation:} mathematically, it meas that
    % $$
    % Y_i(T_i) = Y_i(T_1, \dots, Y_n),
    % $$
    % or 
    % $$
    % Y_i \perp \lrc{T_i : i\neq j},
    % $$
    % or
    $$
    \Pr[Y_i \leq y\ |\ T_1, \dots, T_n] = \Pr[Y_i \leq y\ |\ T_i].
    $$
\end{itemize}

\textbf{Example:}
\begin{itemize}
    \item \textbf{Scenario:} Comparing callback rates for résumés with black-sounding names (treatment) vs. white-sounding names (control).
    \item \textbf{Compliance case:} an individual’s callback decision does not depend on the name on other résumés.
    \item \textbf{Non-compliance case:} an employer who receives several résumés with white-sounding names may become more biased and less likely to call presumably back applicants.
\end{itemize}

\end{frame}

\begin{frame}{What would happen without the NI assumption. Accounting for interaction}

    \begin{itemize}
        \item \textbf{Without NI:} You may need to account for cross interaction
        \item \textbf{Example:} Suppose that 
        \begin{itemize}
            \item[$\bullet$] The résumés are sent in pairs to the same employer. 
            \item[$\bullet$] You want to estimate the callback average difference between sending black-black vs white-black sounding names.
        \end{itemize}
        $$
        \Esp[Y_1 | T_1 = 1, T_2 = 1] - \Esp[Y_1 | T_1 = 1, T_2 = 0].
        $$
        \item \textbf{Drawbacks:} 
        \begin{itemize}
            \item[$\bullet$] The conclusion is less general.
            \item[$\bullet$] Requires higher dimensional data.
        \end{itemize}
    \end{itemize}
      
\end{frame}

\begin{frame}{What would happen without the NI assumption. Layering }

    \begin{itemize}
        \item \textbf{Without NI:} You may need to consider groups where NI hold
        \item \textbf{Example:} You want to test if reduced class sizes causes improved student performance in mathematics. 
        \begin{itemize}
            \item[$\bullet$] \textbf{Problem:} It is not sensible to assume that there is no interference among students within the same school or classroom.
            \item[$\bullet$] \textbf{Solution:} Assume NI between different school/classrooms. 
        \end{itemize}
        
        \item \textbf{Drawbacks:} 
        \begin{itemize}
            \item[$\bullet$] The conclusion is less general.
            \item[$\bullet$] Requires higher data sizes.
        \end{itemize}
    \end{itemize}
      
\end{frame}

\begin{frame}{The SUTVA: no hidden variants}

\begin{itemize}
    \item \textbf{Interpretation:} Under the NHV assumption, the possible implementations of the treatment are irrelevant for the outcome of one individual. That is, \textbf{the outcome depends only on the treatment level} ($0$ or $1$ for binary treatments) and not on how this treatment was implemented. 
    \item \textbf{Mathematical interpretation:} mathematically, this assumption means that the outcome is a (random) function of the treatment
    $$
    Y_i = Y_i(T_i).
    $$
    For binary treatments: 
    $$
    Y_i =
    \begin{cases}
        Y_i(0), & \text{if } T_i = 0 \\
        Y_i(1), & \text{if } T_i = 1
    \end{cases}.
    $$
\end{itemize}

\end{frame}
    
\begin{frame}{The SUTVA: no hidden variants. An example}

\begin{itemize}
    \item \textbf{Scenario:} Comparing callback rates for résumés with black-sounding names (treatment) vs. white-sounding names (control).
    \item \textbf{Compliance case:} All résumés in the treatment group and control group are identical, only varying the name. 
    
    \item \textbf{Non-compliance case:} Some résumés with black-sounding names are different in format \textcolor{deepskyblue3}{(e.g., adding extra details, changing the style)}, or vary significantly in education levels, which might lead to different outcomes.
\end{itemize}

\end{frame}

\begin{frame}{What would happen without the NHV assumption?}

\begin{itemize}
    \item The NHV assumption allows to identify the observed outcome as the outcome under or without the treatment. 

    \item Without the NHV assumption we just observe outcome attached to the specific version of the treatment that was applied.
\end{itemize}

\begin{frame}{Potential outcomes, factual and counterfactual. Under NHV}
\vspace{-0.4cm}

\begin{center}
    \textbf{Under the NHV assumption:}
\end{center}

\begin{itemize}
    \item \textbf{Potential Outcomes:}
    \begin{itemize}
        \item[$\bullet$] $\pmb{Y(1)}$\textbf{:} Outcome under treatment. \textcolor{deepskyblue3}{(e.g, callback if  black-named applicant)}
        \vspace{0.1cm}
        
        \item[$\bullet$] $\pmb{Y(0)}$\textbf{:} Outcome without treatment. \textcolor{deepskyblue3}{(e.g., callback if white-named applicant)}
    \end{itemize}

    \item \textbf{Factual outcome:} Observable outcome (what happened). 
    $$Y(T)$$
    
    \item \textbf{Counterfactual outcome:} Unobservable outcome (what would have happened if the treatment were different). 
    $$\quad Y(1-T)\quad \text{(for binary treatments)}.$$
\end{itemize}

\end{frame}

\begin{frame}{Potential outcomes, factual and counterfactual. Without NHV}
\vspace{-0.4cm}

\begin{center}
    \textbf{Without the NHV assumption:}
\end{center}

\begin{itemize}
    \item \textbf{Potential Outcomes:}
    \begin{itemize}
        \item[$\bullet$] $\pmb{Y(1, j)}$\textbf{:} Outcome under variation $j$ of the treatment, for $j = 1,\dots,v$.
        \vspace{0.1cm}
        
        \item[$\bullet$] $\pmb{Y(0, j)}$\textbf{:} Outcome without variation $j$ of the treatment, for $j = 1,\dots, v$.
    \end{itemize}

    \item \textbf{Factual outcome:} Observable outcome (what happened). 
    $$Y(T, j)$$
    
    \item \textbf{Counterfactual outcome:} Unobservable outcome (what would have happened). 
    $$\quad Y(1-T, j)\quad \text{(for binary treatments)}.$$
\end{itemize}

\textbf{Conclusion:} You will need increasingly higher sample size as the number of variants increases. Impracticable for infinite number of variants.

\end{frame}

\section{\Large{The Potential Outcome Framework}}

\begin{frame}{The Potential Outcome Framework. Introduction and origins}

\begin{coolblock}{The Potential Outcome Framework\footnote{\scriptsize Donald B Rubin (2006) Causal Inference Using Potential Outcomes: Design, Modeling, Decisions.}}
    The \textbf{Potential Outcome Framework} (POF), also known as the Rubin Causal Model, is a widely used approach in \textbf{causal inference} that helps to understand and estimate \textbf{causal effects}, based on the idea of comparing \textbf{potential outcomes}.
\end{coolblock}
\vspace{0.2cm}

\textbf{Rubin's definition of causal effect:} ``Intuitively, the causal effect of one treatment, $E$, over another, $C$, is the difference between \textbf{what would have happened if} the unit had been exposed to $E$ and what would have happened if the unit had been exposed to $C$'' 
\vspace{0.3cm}

\textbf{Rubin's example of potential outcomes:} ``If an hour ago I had taken two aspirins instead of just a glass of water, my headache would now be gone, or because an hour ago I took two aspirins instead of just a glass of water, my headache is now gone.''

\end{frame}

\begin{frame}{Individual causal effect and the fundamental problem of causal inference}

%  where $Y_i(1)$ represents the outcome that would be realized under the treatment condition ($T_i = 1$) and $Y_i(0)$ denotes the outcome that would be realized under the control condition ($T_i = 0$).

\begin{itemize}
    \item \textbf{Individual treatment (causal) effect:} difference between the two potential outcomes: $$\mathrm{TE} = Y(1) - Y(0).$$
\end{itemize}

\begin{coolblock}{Fundamental Problem of Causal Inference\footnote{\scriptsize Paul W. Holland (1985) Statistics and Causal Inference.} (FPCI)}
    We observe only one of the two potential outcomes (the factual outcome), so we never get to observe the effect of the treatment on the outcome.
\end{coolblock}
\vspace{0.2cm}

\begin{itemize}
    \item \textbf{Individual causal inference:} The FPCI states that it is impossible to estimate the individual causal effect. You only have one non-observable observation for this phenomenon!
\end{itemize}

\end{frame}

\begin{frame}{Average causal effect}

\begin{itemize}
    \item \textbf{Average causal inference:} Despite the FPCI, it is possible to estimate the causal effect on average terms.

    \item \textbf{Average treatment (causal) effect:} Aggregated measure of the treatment effect across all the individuals in a population. Defined as the expectation of the individual treatment effect: 
    $$
    \mathrm{ATE} = \Esp[\mathrm{TE}] = \Esp[Y(1) - Y(0)].
    $$
\end{itemize}

\begin{coolblock}{The Sample Average Treatment Effect (SATE)}
    The \textbf{sample average treatment effect} is the sample average of individual-level causal effects:
    $$
    \mathrm{SATE} = \oline{\mathrm{TE}} = \frac{1}{n}\sum_{i=1}^n \mathrm{TE}_i = \frac{1}{n}\sum_{i=1}^n \lrp{Y_i (1) - Y_i(0)} = \oline{Y(1)} - \oline{Y(0)},
    $$
    where $n$ is the sample size and the subindex $i$ represents the $i$th observation.
\end{coolblock}
    
\end{frame}

\begin{frame}{Estimating the SATE}

    Depending on the \textbf{data type} and its \textbf{underlying circumstances}, many methods can be used to \textbf{estimate the SATE}. Among the most common ones are:
    \vspace{0.2cm}
    
    \begin{itemize}
        \item Difference-in-Means (DiM) estiamator
        \item Before-and-after (BA) estimator
        \item Difference-in-differences (DiD) estimator
    \end{itemize}
    \vspace{0.2cm}

    \textbf{Choosing the right estimator} (next chapter) is critical to \textbf{avoid bias in the estimation}.
        
\end{frame}

\begin{frame}{Pretreatment variables and Confounders}

    \textbf{Pretreatment variable:} measured before the treatment or intervention is applied \textcolor{deepskyblue3}{(e.g., age, education, gender,...)}
    
    \begin{coolblock}{Confounders\footnote{\scriptsize \textbf{Alternative names:} confounding factor, common causes.}}
        A pretreatment variable that is associated with both the treatment and the outcome variables is called a confounder and is a \textbf{source of confounding bias} in the estimation of the treatment effect.
    \end{coolblock}
    
\end{frame}

\begin{frame}{Confounding bias}
    
    \begin{coolblock}{Confounding bias}
       Distortion of the estimated relationship between the treatment and the outcome due to the influence of a confounder
    \end{coolblock}

    \textbf{Example:} 
    \begin{itemize}
        \item \textbf{Correlation with the treatment:} the résumés of white-sounding names may be associated with higher education status.
        \item \textbf{Correlation with the outcome:} higher education status may positively influence callback rates.
        \item \textbf{Confounder:} Education status may be a confounder.
    \end{itemize}

    % \textbf{How to address:}
    % \begin{itemize}
    %     \item \textbf{Random Sampling:} Ensures similar proportion of confounders among the treatment and control groups.
    %     % \item \textbf{Weighting Adjustments:} Adjusting the analysis to account for the overrepresentation or underrepresentation of confounders.
    % \end{itemize}
    
\end{frame}

\begin{frame}{Selection bias}
    
    \begin{coolblock}{Selection bias}
       Phenomenon that occurs when the sample of individuals selected for a study is not representative of the broader population, leading to biased estimates.
    \end{coolblock}

    \textbf{Example:} 
    \begin{itemize}
        \item \textbf{Non-representative sample:} Sending only résumés with high levels of education. 
        \item \textbf{Selection bias:} the results may not reflect the discrimination faced by less-educated individuals. 
    \end{itemize}

    % \textbf{How to address:}
    % \begin{itemize}
    %     \item \textbf{Random Sampling:} Ensure that the study includes a diverse range of job applicants, representing various education levels.
    %     \item \textbf{Stratification:} Group applicants by factors like education, ensuring that each group is proportionally represented in the study to account for selection bias.
    % \end{itemize}

\end{frame}

\begin{frame}{Types of Selection Bias}

\begin{itemize}
    \item \textbf{Sampling Bias:} When certain individuals are more likely to be selected, leading to a non-representative sample.
    \item \textbf{Self-Selection Bias:} When participants choose themselves into a study, potentially skewing results.
    \item \textbf{Attrition Bias:} When participants drop out of a study, affecting the representativeness of the remaining sample.
    \item \textbf{Non-Response Bias:} When those who don't respond differ systematically from those who do.
    \item \textbf{Time Interval Bias:} When data collection timing affects the results.
    \item \textbf{Survivorship Bias:} When only successful subjects are considered, overlooking failures.
    \item \textbf{Common Source Bias:} When both variables are collected from the same group, causing shared biases.
\end{itemize}
    
\end{frame}
