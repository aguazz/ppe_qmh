\begin{enumerate}
    \item[(A)] \( P(A \cap B) = P(A) \cdot P(B) \)
    \item[(B)] \( P(A \cap B) \neq P(A) \cdot P(B) \)
    \item[(C)] Having the information that one already happened does not change the probability of the other one
    \item[(D)] \( P(H \mid M) \neq P(H) \)
\end{enumerate}
