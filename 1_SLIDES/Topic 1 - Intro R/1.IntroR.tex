\begin{frame}{Style Guide (by \href{http://adv-r.had.co.nz/Style.html}{Hadley Wickham})}
\small

\begin{enumerate}
    \item Use meaningful file names ending in \texttt{.R} (e.g., \texttt{cleaning.R}, \texttt{regression.R})\smallskip
    \item Use \texttt{lowercase\_with\_underscores} for names (e.g., \texttt{turnout\_rate}, \texttt{load\_data()})\smallskip
    \item Variable names and commands are case-sensitive (e.g., \texttt{mean} $\neq$ \texttt{Mean}; \texttt{x} $\neq$ \texttt{X})\smallskip
    \item Names may include letters, numbers (not first), and dots (e.g., \texttt{x2}, \texttt{party.name}; not \texttt{2x})\smallskip
    \item Don’t reuse names of existing functions/objects\smallskip
    \item Put spaces around operators (e.g., \texttt{x <- 1}, \texttt{y = x + 2})\smallskip
    \item Put a space after commas, never before (e.g., \texttt{c(1, 2, 3)})\smallskip
    \item No spaces inside \texttt{()} or \texttt{[]} (except after commas): \texttt{mean(x)}, \texttt{x[1, 2]}\smallskip
    \item Keep lines $\le$ 80 characters (split long lines if needed)\smallskip
    \item Use \texttt{<-} (not \texttt{=}) for assignment: \texttt{x <- 1}\smallskip
    \item Comment your code: start with \texttt{\#} and keep comments helpful, not excessive
\end{enumerate}
\end{frame}

\section{\Large Basics}

\begin{frame}{Help}
\begin{itemize}
    \item To obtain details about a command, type \newline
    \texttt{?command\_name} \newline
    or \newline
    \texttt{help(command\_name)}
    \item Example: \texttt{?mean} or \texttt{help(mean)}\smallskip
    % \item Start the "help center": \texttt{help.start()} 
    \item In R-Studio: bottom right corner \smallskip
    % \item Task Views on CRAN
    % \item R Journal on CRAN
\end{itemize}
\end{frame}

\begin{frame}{Packages}

One of the strengths of R is the large and growing collection of packages that can be installed from CRAN (The Comprehensive R Archive Network)\smallskip

\begin{itemize}
    \item \textbf{Installation:} \texttt{install.packages(`package\_name')}\smallskip
    \item \textbf{Activation:} \texttt{library(package\_name)}\smallskip
    \item \textbf{Help:} \texttt{library(help=package\_name)}\smallskip
    \item In R-Studio: bottom right corner\smallskip
\end{itemize}
\end{frame}

\begin{frame}{R objects}
\begin{itemize}
    \item R is object-oriented \smallskip
    \item Objects store data, functions, results of operations, ...\smallskip
    \item Everything in R is an object (e.g., scalar, vector, matrix, string)\smallskip
    % \item The object type determines how some commands work (e.g., \texttt{plot}, \texttt{summary})
    \item Every object has a unique (case-sensitive) name\smallskip
    % \item List of all objects: \texttt{ls()}
    % \item Delete (\textbf{r}e\textbf{m}ove) objects: \texttt{rm()}
    \item Examples:
    \begin{itemize}
        \item[$\bullet$] Vectors: \\
            \texttt{x <- c(1, 2, 3)}
        \item[$\bullet$] Matrices: \\
            \texttt{m <- matrix(1:6, nrow = 2, ncol = 3)}
        \item[$\bullet$] Data Frames: \\
            \texttt{df <- data.frame(Name = c("Alice", "Bob"), Age = c(25, 30))}
    \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}{R classes}
    \begin{itemize}
        \item Classes define the type and behavior of an object.\smallskip
        \item Determines how an object responds to functions.\smallskip
        \item Common classes:
        \begin{itemize}
            \item[$\bullet$] \textbf{Numeric:} Numbers (e.g., $3.14, 42$).
            \item[$\bullet$] \textbf{Character:} Text strings (e.g., "Hello").
            \item[$\bullet$] \textbf{Logical:} Boolean values (TRUE, FALSE).
            \item[$\bullet$] \textbf{Factors:} Categorical data with levels.
            \item[$\bullet$] \textbf{Lists:} Collections of objects.\smallskip
        \end{itemize}
        \item Examples:
        \begin{itemize}
        \item[$\bullet$] \textbf{Numeric class:} \\
            \texttt{num <- 42} \\
            \texttt{class(num)}
        \item[$\bullet$] \textbf{Factor class:} \\
            \texttt{gender <- factor(c("male", "female", "female"))} \\
            \texttt{class(gender)}
    \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Functions}

    \begin{itemize}
        \item Functions in R perform specific tasks or operations.\smallskip
        \item \textbf{Syntax:} \texttt{func\_name(input)}\smallskip
        % \begin{itemize}
        %     \item[$\bullet$] \texttt{func\_name} is the function name
        %     \item[$\bullet$] \texttt{input} is the argument or input object.
        % \end{itemize}
        \item Some basic functions:
        \begin{itemize}
            \item[$\bullet$] \texttt{sqrt()} - Square root.
            \item[$\bullet$] \texttt{print()} - Print output.
            \item[$\bullet$] \texttt{c()} - Combine elements into a vector.
            \item[$\bullet$] \texttt{length()} - Number of elements in a vector.
            \item[$\bullet$] \texttt{min()} - Minimum value.
            \item[$\bullet$] \texttt{max()} - Maximum value.
            \item[$\bullet$] \texttt{range()} - Range of values.
            \item[$\bullet$] \texttt{mean()} - Average of values.
            \item[$\bullet$] \texttt{sum()} - Total of values.\smallskip
        \end{itemize}
        \item Example (mean world population from 1950 to 2010): \\
        \small \texttt{world.pop <- c(2525779, 3026003, 3691173, 4449049, 5320817, 6127700, 6916183)} \\
        \texttt{mean(world.pop)}
    \end{itemize}

\end{frame}

\begin{frame}{Functions with multiple arguments}

    \begin{itemize}
        \item \textbf{Syntax:} \texttt{funcname(arg1 = input1, arg2 = input2)}\smallskip
        \item \textbf{Example:} \texttt{seq()} function\smallskip
        \begin{itemize}
            \item[$\bullet$] Generates a sequence of numbers.\smallskip
            \item[$\bullet$] Arguments: \\
            \texttt{from} - start value \\
            \texttt{to} - end value \\
            \texttt{by} - interval step value\smallskip
        \end{itemize}
        \item \textbf{Implementation:} \texttt{seq(from = 1950, to = 2010, by = 10)}
    \end{itemize}
    
\end{frame}

\begin{frame}{Managing working directory}

    \begin{itemize}
        \item The working directory is where R reads and saves files.\smallskip
        \item Check the current directory: \\
        \texttt{getwd()}\smallskip
        \item Set a new directory via the console: \\
        \texttt{setwd("path/to/folder")}\smallskip
        \item Set a new directory via the menu bar: \\
        Session $\rightarrow$ Set Working Directory $\rightarrow$:
        \begin{itemize}
            \item[$\bullet$] To Source File Location (current file path)
            \item[$\bullet$] Choose Directory... (custom path)
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Loading data}

    \begin{itemize}
        \item \textbf{CSV files:} tabular data with observations separated by line breaks and variables within the observation separated by a comma, a tab, or other characters.\smallskip
        \item \textbf{Read CSV files:} \texttt{read.csv(path/to/file)}\smallskip
        \item Other formats and functions to read them:
        \begin{itemize}
            \item[$\bullet$] Excel Files: \texttt{read\_excel("file.xlsx")} (requires \texttt{readxl} package)
            \item[$\bullet$] RData Files: \texttt{load("file.RData")}
        \end{itemize}
    \end{itemize}
    
\end{frame}

\begin{frame}{Storing and exploring data}

    \begin{itemize}
        \item \textbf{Storing Data:} typically stored in \texttt{data.frame} objects\smallskip
        \item \textbf{Summary function:}
        \begin{itemize}
            \item[$\bullet$] \texttt{names()} - Names of the variables (columns)
            \item[$\bullet$] \texttt{nrow()} - Number of rows
            \item[$\bullet$] \texttt{ncol()} - Number of columns
            \item[$\bullet$] \texttt{dim()} - Dimensions
            \item[$\bullet$] \texttt{summary()} - Summary statistics
        \end{itemize}
            
    \end{itemize}
    
\end{frame}

\begin{frame}{Accessing variables and observations}

    Imaging that the object \texttt{UN.pop} is a data.frame containing 7 observations (rows) of the variables (columns) \texttt{year} and \texttt{world.pop}. You can access specific variables and observations like this:
    \begin{itemize}
        \item \textbf{Using the \$ operator:} \\
        \texttt{UN.pop\$world.pop} - extract the \texttt{world.pop} variable
        \item \textbf{Using [ ]:} \\
        \texttt{UN.pop[, "world.pop"]} - extract the \texttt{world.pop} column \\
        \texttt{UN.pop[c(1, 2, 3), ]} - extract the first three rows (all columns) \\
        \texttt{UN.pop[1:3, "year"]} - extract the first three values of the \texttt{year} column
        \item \textbf{Combining techniques to retrieve specific elements:} \\
        \texttt{UN.pop\$world.pop[seq(from = 1, to = nrow(UNpop), by = 2)]} -  retrieve odd elements of the \texttt{world.pop} variable
    \end{itemize}
\end{frame}
